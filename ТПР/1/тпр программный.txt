-------------------Написать и отладить линейную программу-сценарий---------------
% Ввод чисел
a = input('Введите число a: ');
b = input('Введите число b: ');

% Ввод вектора
vector = input('Введите вектор в виде [x1, x2, x3, ...]: ');

% Ввод матрицы
matrix = input('Введите матрицу в виде [x1, x2, x3; x4, x5, x6; x7, x8, x9]: ');

% Счет
result = a + b;
vector_result = vector * 2;
matrix_result = matrix + 10;

% Вывод результатов
disp(['Результат сложения чисел a и b: ', num2str(result)]);
disp('Результат умножения вектора на 2:');
disp(vector_result);
disp('Результат прибавления 10 к матрице:');
disp(matrix_result);
-------------------Написать и отладить разветвленную программу-сценарий-------------
% Оператор if с тремя условиями
x = input('Введите число x: ');
if x < 0
    disp('x меньше нуля');
elseif x == 0
    disp('x равно нулю');
else
    disp('x больше нуля');
end

% Оператор switch с тремя выборами
choice input('Введите букву A, B или C ');

switch choice
    case 'A'
        disp('Выбран вариант A');
    case 'B'
        disp('Выбран вариант B');
    case 'C'
        disp('Выбран вариант C');
    otherwise
        disp('Выбран неизвестный вариант');
end
--------------------Написать и отладить циклическую программу-сценарий для вывода матрицы поэлементно с комментариями------------
matrix = magic(3); % Пример матрицы

[rows, cols] = size(matrix);

for i = 1:rows
    for j = 1:cols
        fprintf('Элемент в строке %d и столбце %d: %d\n', i, j, matrix(i, j));
    end
end
---------------Написать и отладить замкнутую программу-сценарий------------
while true
    user_input = input('Введите что-нибудь (для выхода введите "exit"): ', 's');
    if strcmpi(user_input, 'exit')
        break;
    end
    disp(['Вы ввели: ', user_input]);
end

% Функция для ввода и вывода данных с комментариями
function inputOutputWithComments()
    % Ввод числа с комментарием
    num = input('Введите число: ');
    % Вывод числа с комментарием
    fprintf('Вы ввели число: %f\n', num);
    % Ввод строки с комментарием
    str = input('Введите строку: ', 's');
    % Вывод строки с комментарием
    fprintf('Вы ввели строку: %s\n', str);
end
